## 七、附录：分析 SQL 清单

### 7.1 时间戳解析
```sql
-- 毫秒 → 可读时间
SELECT FROM_UNIXTIME(ts / 1000) FROM disk_tsar;

-- 毫秒 → 截断到整点小时
SELECT FROM_UNIXTIME(ts / 1000, '%Y-%m-%d %H:00') FROM pref_tsar;

-- 数据时间跨度
SELECT FROM_UNIXTIME(MIN(ts)/1000), FROM_UNIXTIME(MAX(ts)/1000),
       DATEDIFF(FROM_UNIXTIME(MAX(ts)/1000), FROM_UNIXTIME(MIN(ts)/1000))
FROM disk_tsar;
```

### 7.2 按小时汇总
```sql
-- 磁盘指标按小时汇总
SELECT hostid, `mod`,
    FROM_UNIXTIME(ts/1000, '%Y-%m-%d %H:00') AS hour,
    COUNT(*) AS samples, ROUND(AVG(value),2) AS avg_val, ROUND(MAX(value),2) AS max_val
FROM disk_tsar
GROUP BY hostid, `mod`, hour;

-- 性能指标 + 维度关联
SELECT h.hostname, m.`desc`, m.unit,
    FROM_UNIXTIME(p.ts/1000, '%Y-%m-%d %H:00') AS hour,
    COUNT(*) AS samples, ROUND(AVG(p.value),2) AS avg_val
FROM pref_tsar p
JOIN host_detail h ON p.hostid = h.hostid
JOIN mod_detail m ON p.`mod` = m.`mod`
GROUP BY h.hostname, m.`desc`, m.unit, hour;
```

### 7.3 ER 关系与外键
```sql
-- 外键定义
ALTER TABLE disk_tsar ADD CONSTRAINT fk_disk_host FOREIGN KEY (hostid) REFERENCES host_detail(hostid);
ALTER TABLE disk_tsar ADD CONSTRAINT fk_disk_mod  FOREIGN KEY (`mod`)  REFERENCES mod_detail(`mod`);
ALTER TABLE pref_tsar ADD CONSTRAINT fk_pref_host FOREIGN KEY (hostid) REFERENCES host_detail(hostid);
ALTER TABLE pref_tsar ADD CONSTRAINT fk_pref_mod  FOREIGN KEY (`mod`)  REFERENCES mod_detail(`mod`);

-- 完整性验证（期望 0）
SELECT COUNT(*) FROM disk_tsar d LEFT JOIN host_detail h ON d.hostid = h.hostid WHERE h.hostid IS NULL;
SELECT COUNT(*) FROM pref_tsar p LEFT JOIN mod_detail m ON p.`mod` = m.`mod` WHERE m.`mod` IS NULL;
```
